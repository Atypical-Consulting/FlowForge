```typescript
import { DiffEditor, loader } from '@monaco-editor/react';
import { useQuery } from '@tanstack/react-query';
import { commands } from '../../bindings';
import { useStagingStore } from '../../stores/staging';
import { useState } from 'react';
import { Columns, AlignJustify, Loader2 } from 'lucide-react';
import { Button } from '../ui/button';

// Configure Monaco to use local files instead of CDN (optional, for offline)
// loader.config({ paths: { vs: '/monaco-editor/min/vs' } });

export function DiffViewer() {
  const { selectedFile } = useStagingStore();
  const [inline, setInline] = useState(true);
  const [contextLines] = useState(3);
  
  // Determine if selected file is staged or unstaged
  // For now, default to unstaged (workdir diff)
  const staged = false; // TODO: Track which section file came from
  
  const { data: diff, isLoading, error } = useQuery({
    queryKey: ['fileDiff', selectedFile?.path, staged, contextLines],
    queryFn: () => selectedFile 
      ? commands.getFileDiff({ 
          path: selectedFile.path, 
          staged, 
          contextLines 
        })
      : null,
    enabled: !!selectedFile,
  });
  
  if (!selectedFile) {
    return (
      <div className="flex-1 flex items-center justify-center bg-gray-900">
        <p className="text-gray-500 text-sm">
          Select a file to view diff
        </p>
      </div>
    );
  }
  
  if (isLoading) {
    return (
      <div className="flex-1 flex items-center justify-center bg-gray-900">
        <Loader2 className="w-5 h-5 animate-spin text-gray-400" />
      </div>
    );
  }
  
  if (error || !diff) {
    return (
      <div className="flex-1 flex items-center justify-center bg-gray-900">
        <p className="text-red-400 text-sm">
          Failed to load diff
        </p>
      </div>
    );
  }
  
  if (diff.isBinary) {
    return (
      <div className="flex-1 flex items-center justify-center bg-gray-900">
        <p className="text-gray-400 text-sm">
          Binary file - diff not available
        </p>
      </div>
    );
  }
  
  return (
    <div className="flex-1 flex flex-col overflow-hidden">
      {/* Toolbar */}
      <div className="flex items-center gap-2 px-3 py-2 border-b border-gray-800 bg-gray-950">
        <span className="text-sm text-gray-300 truncate flex-1">
          {selectedFile.path}
        </span>
        <Button
          variant="ghost"
          size="sm"
          onClick={() => setInline(!inline)}
          title={inline ? 'Switch to side-by-side' : 'Switch to inline'}
        >
          {inline ? (
            <Columns className="w-4 h-4" />
          ) : (
            <AlignJustify className="w-4 h-4" />
          )}
        </Button>
      </div>
      
      {/* Monaco DiffEditor */}
      <div className="flex-1">
        <DiffEditor
          original={diff.oldContent}
          modified={diff.newContent}
          language={diff.language}
          theme="vs-dark"
          options={{
            readOnly: true,
            renderSideBySide: !inline,
            originalEditable: false,
            automaticLayout: true,
            scrollBeyondLastLine: false,
            minimap: { enabled: false },
            fontSize: 13,
            lineNumbers: 'on',
            folding: true,
            wordWrap: 'off',
          }}
        />
      </div>
    </div>
  );
}
```

Note: The `language` prop comes from the backend's file extension detection. Monaco supports: javascript, typescript, typescriptreact, json, markdown, rust, python, css, html, etc.

**Create `src/components/diff/HunkHeader.tsx`:**

Per 02-RESEARCH.md, hunk-level staging is NOT natively supported by libgit2 and requires manual blob manipulation. This task creates the UI structure with disabled buttons, ready for future backend implementation.

```typescript
import { Plus, Minus, ChevronDown, ChevronRight } from 'lucide-react';
import { Button } from '../ui/button';
import { cn } from '../../lib/utils';
import { useState } from 'react';

interface HunkHeaderProps {
  hunkIndex: number;
  startLine: number;
  lineCount: number;
  header: string; // e.g., "@@ -10,5 +10,7 @@ function example()"
  section: 'staged' | 'unstaged';
  onStageHunk?: (hunkIndex: number) => void;
  onDiscardHunk?: (hunkIndex: number) => void;
  disabled?: boolean; // True until backend supports hunk staging
}

export function HunkHeader({ 
  hunkIndex,
  startLine,
  lineCount,
  header,
  section,
  onStageHunk,
  onDiscardHunk,
  disabled = true, // Disabled by default - backend not ready
}: HunkHeaderProps) {
  const [collapsed, setCollapsed] = useState(false);
  
  // Parse function context from header if present
  const functionMatch = header.match(/@@.*@@\s*(.*)/);
  const functionContext = functionMatch?.[1] || '';
  
  return (
    <div className={cn(
      'flex items-center gap-2 px-3 py-1.5 bg-gray-800/70 border-y border-gray-700',
      'text-xs font-mono text-gray-400'
    )}>
      {/* Collapse toggle */}
      <button
        onClick={() => setCollapsed(!collapsed)}
        className="p-0.5 hover:bg-gray-700 rounded"
      >
        {collapsed ? (
          <ChevronRight className="w-3 h-3" />
        ) : (
          <ChevronDown className="w-3 h-3" />
        )}
      </button>
      
      {/* Line info */}
      <span className="text-blue-400">
        @@ {startLine},{lineCount} @@
      </span>
      
      {/* Function context */}
      {functionContext && (
        <span className="text-gray-500 truncate flex-1">
          {functionContext}
        </span>
      )}
      
      {/* Stage/Unstage hunk button */}
      {section === 'unstaged' && (
        <Button
          variant="ghost"
          size="sm"
          onClick={() => onStageHunk?.(hunkIndex)}
          disabled={disabled}
          title={disabled ? "Hunk staging coming soon" : "Stage this hunk"}
          className="h-5 px-1.5 text-xs"
        >
          <Plus className="w-3 h-3 mr-1" />
          Stage
        </Button>
      )}
      
      {section === 'staged' && (
        <Button
          variant="ghost"
          size="sm"
          onClick={() => onDiscardHunk?.(hunkIndex)}
          disabled={disabled}
          title={disabled ? "Hunk unstaging coming soon" : "Unstage this hunk"}
          className="h-5 px-1.5 text-xs"
        >
          <Minus className="w-3 h-3 mr-1" />
          Unstage
        </Button>
      )}
    </div>
  );
}

// Export collapsed state for parent to conditionally hide hunk content
export function useHunkCollapse() {
  return useState<Set<number>>(new Set());
}
```

```typescript
import { useState } from 'react';
import { useMutation, useQueryClient, useQuery } from '@tanstack/react-query';
import { commands } from '../../bindings';
import { Button } from '../ui/button';
import { cn } from '../../lib/utils';
import { Loader2, RotateCcw } from 'lucide-react';

export function CommitForm() {
  const [message, setMessage] = useState('');
  const [amend, setAmend] = useState(false);
  const queryClient = useQueryClient();
  
  // Get staging status to check if there are staged files
  const { data: status } = useQuery({
    queryKey: ['stagingStatus'],
    queryFn: () => commands.getStagingStatus(),
  });
  
  const commitMutation = useMutation({
    mutationFn: () => commands.createCommit({ message, amend }),
    onSuccess: (result) => {
      setMessage('');
      setAmend(false);
      // Invalidate staging status and history
      queryClient.invalidateQueries({ queryKey: ['stagingStatus'] });
      queryClient.invalidateQueries({ queryKey: ['commitHistory'] });
      queryClient.invalidateQueries({ queryKey: ['repositoryStatus'] });
    },
  });
  
  const hasStagedFiles = status && status.staged.length > 0;
  const canCommit = hasStagedFiles && message.trim().length > 0;
  
  // Parse subject and body
  const lines = message.split('\n');
  const subject = lines[0] || '';
  const subjectLength = subject.length;
  
  // Subject line guidance
  const subjectStatus = 
    subjectLength === 0 ? 'empty' :
    subjectLength <= 50 ? 'good' :
    subjectLength <= 72 ? 'warning' :
    'error';
  
  return (
    <div className="border-t border-gray-800 p-3 bg-gray-950">
      <div className="space-y-2">
        <textarea
          value={message}
          onChange={(e) => setMessage(e.target.value)}
          placeholder="Commit message..."
          className={cn(
            'w-full h-24 px-3 py-2 text-sm bg-gray-900 border border-gray-700',
            'rounded resize-none focus:outline-none focus:border-blue-500',
            'text-gray-200 placeholder:text-gray-500'
          )}
        />
        
        {/* Character count and guidance */}
        <div className="flex items-center justify-between text-xs">
          <div className="flex items-center gap-2">
            <label className="flex items-center gap-1.5 text-gray-400">
              <input
                type="checkbox"
                checked={amend}
                onChange={(e) => setAmend(e.target.checked)}
                className="rounded border-gray-600"
              />
              Amend last commit
            </label>
          </div>
          
          <span className={cn(
            'font-mono',
            subjectStatus === 'good' && 'text-green-500',
            subjectStatus === 'warning' && 'text-yellow-500',
            subjectStatus === 'error' && 'text-red-500',
            subjectStatus === 'empty' && 'text-gray-500'
          )}>
            {subjectLength}/50
            {subjectStatus === 'warning' && ' (suggested max)'}
            {subjectStatus === 'error' && ' (too long)'}
          </span>
        </div>
        
        {/* Commit button */}
        <Button
          onClick={() => commitMutation.mutate()}
          disabled={!canCommit || commitMutation.isPending}
          className="w-full"
        >
          {commitMutation.isPending ? (
            <>
              <Loader2 className="w-4 h-4 mr-2 animate-spin" />
              Committing...
            </>
          ) : amend ? (
            <>
              <RotateCcw className="w-4 h-4 mr-2" />
              Amend Commit
            </>
          ) : (
            'Commit'
          )}
        </Button>
        
        {!hasStagedFiles && (
          <p className="text-xs text-gray-500 text-center">
            No staged changes to commit
          </p>
        )}
        
        {commitMutation.isError && (
          <p className="text-xs text-red-400 text-center">
            {String(commitMutation.error)}
          </p>
        )}
      </div>
    </div>
  );
}
```
